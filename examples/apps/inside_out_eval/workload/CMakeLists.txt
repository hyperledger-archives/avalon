CMAKE_MINIMUM_REQUIRED(VERSION 3.2 FATAL_ERROR)

FILE(GLOB PROJECT_HEADERS "*.h")
FILE(GLOB PROJECT_SOURCES "*.cpp")

SET(GENERIC_CXX_FLAGS ${DEBUG_FLAGS} "-Wall" "-fPIC" "-Wno-write-strings" "-std=c++11")

# Workload static library name must match the name of the workload folder
# created under $TCF_HOME/example/apps/<workload_name>
SET(INSIDE_OUT_EVAL_STATIC_NAME inside_out_eval)
PROJECT(${INSIDE_OUT_EVAL_STATIC_NAME} C CXX)

ADD_LIBRARY(${INSIDE_OUT_EVAL_STATIC_NAME} STATIC ${PROJECT_HEADERS} ${PROJECT_SOURCES})

TARGET_INCLUDE_DIRECTORIES(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE ${TCF_TOP_DIR}/tc/sgx/trusted_worker_manager/enclave)
TARGET_INCLUDE_DIRECTORIES(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE ${GENERIC_PRIVATE_INCLUDE_DIRS})
TARGET_INCLUDE_DIRECTORIES(${INSIDE_OUT_EVAL_STATIC_NAME} PUBLIC ${GENERIC_PUBLIC_INCLUDE_DIRS})
TARGET_INCLUDE_DIRECTORIES(${INSIDE_OUT_EVAL_STATIC_NAME} PUBLIC "${SGX_SDK}/include/tlibc")
TARGET_INCLUDE_DIRECTORIES(${INSIDE_OUT_EVAL_STATIC_NAME} PUBLIC "${SGX_SDK}/include/libcxx")

TARGET_COMPILE_OPTIONS(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE "${GENERIC_CXX_FLAGS}")
TARGET_COMPILE_OPTIONS(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE "-nostdinc")
TARGET_COMPILE_OPTIONS(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE "-nostdinc++")
TARGET_COMPILE_OPTIONS(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE "-fno-builtin-printf")

TARGET_COMPILE_DEFINITIONS(${INSIDE_OUT_EVAL_STATIC_NAME} PRIVATE "-DUSE_DL=1")
